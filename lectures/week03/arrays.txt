ARRAYS:

Arrays are used to group variables of the same type under one name.
They are called ELEMENTS of the array, and are referred to with the
array name, and an INDEX VALUE contained within square brackets.

/*-----------------------------------------------------------------
*	array0.cpp	-	introduction to the idea of arrays.
* Mon May 24 16:39:39 PDT 2004 TEK
-----------------------------------------------------------------*/
#include <iostream>

using namespace std ;

int print_array (int x[], int size) ;

int main (int argc, char *argv[], char **env)
{
        int i,  a[5] ;
	char s[82] ;
	// an array is a "homogeneous" data structure:
	// that means, it contains many data items,
	// which all have the same TYPE.
	// don't count on arrays being initialized to 0
	// before use, do it yourself.  The n00b way is with a loop:
	for ( i = 0; i < 5 ; i++)
		a[i] = 0 ;
	print_array(a, 5) ;
	// elements of the array can hold different values:
	a[0] = 456 ;
	a[2] = 7865 ;
	print_array(a,5) ;
	return 0 ;
} // main

int print_array (int x[], int size) 
{
	int i ;
	char s[91] ;
	cout << "=+=+=+=+=+=+=+=+=+=+=+=" << endl ;
	for ( i = 0 ; i < size ; i++) {
		sprintf(s,"x[%d] : %d", i, x[i]) ;
		cout << s << endl ;
	}
	return 0 ;
} // print_array() ends

======================

Arrays and pointers are basically the same thing: names for places in
memory where like-typed items are stored.


/*-----------------------------------------------------------------
*	array1.cpp	-	first look at arrays and pointers
*				after K&R Chapter5.
*	Tue Jul 29 17:38:35 PDT 2003 TEK.
An array is a pointer, a pointer is an address, an address is
an array...they are all just different names for the reality
of an ADDRESS IN THE MACHINE.  You can use an array name like 
a pointer constant, or use a pointer variable like the name 
of an array 
-----------------------------------------------------------------*/
#include <stdio.h>
#include <iostream>

using namespace std ;

int main (int argc, char *argv[], char **env)
{
	int i,  array1[30], 
	*ip0, *ip1, *ip2  ;
	char s[82] ;

	// load big array with consecutive integers.
	for ( i = 0 ; i < 30 ; i++) array1[i] = i ;

	// initialize pointers to segments of the big array
	// it is possible to point pointer variables at parts of
	// a larger array, then treat them as discrete smaller
	// arrays:
	ip0 = array1 ;
	ip1 = &array1[10]; ip2 = &array1[20]; 

	for ( i = 0 ; i < 10 ; i++)
	{
		sprintf(s,"Array1[%d] : %3d\t", i , array1[i]) ; cout << s ;
		sprintf(s,"ip0[%d] : %3d\t", i , ip0[i]) ; cout << s ;
		sprintf(s,"ip1[%d] : %3d\t", i , ip1[i]) ; cout << s ;
		sprintf(s,"ip2[%d] : %3d\t", i , ip2[i]) ; cout << s ;
		sprintf(s,"\n") ; cout << s ;
	} // for
} // main
